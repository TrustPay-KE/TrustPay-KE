Get Current User (on page load)

async function initPage() {
    const user = await getCurrentUser();
    
    if (!user) {
        window.location.href = 'login.html';
        return;
    }
    
    console.log('Welcome', user.name);
}

initPage();

Transaction Examples

Create Transaction

import { createTransaction, calculateServiceFee } from './js/index.js';

// Calculate total
const amount = 5000;
const fee = calculateServiceFee(amount);
const total = amount + fee;

// Create transaction
const result = await createTransaction({
    senderName: 'John Doe',
    senderEmail: 'john@example.com',
    receiverName: 'Jane Smith',
    receiverEmail: 'jane@example.com',
    amount: amount,
    description: 'Website design services',
    type: 'buyer'  // or 'seller'
});

if (result.success) {
    console.log('Tracking:', result.transaction.trackingNumber);
}

Get User Transactions

import { getUserTransactions, getTransactionStats } from './js/index.js';

// Get all transactions
const { transactions, success } = await getUserTransactions();

if (success) {
    transactions.forEach(tx => {
        console.log(`${tx.trackingNumber}: KES ${tx.amount} - ${tx.status}`);
    });
}

// Get statistics
const { stats } = await getTransactionStats();
console.log(`Total: ${stats.total}, Pending: ${stats.pending}`);

M-Pesa Examples

Get Manual Payment Instructions

import { getMpesaManualPayment } from './js/index.js';

const payment = getMpesaManualPayment(5000);

console.log('Paybill:', payment.paybill);
console.log('Account:', payment.accountNumber);
console.log('Instructions:', payment.instructions);

Parse M-Pesa Message (for verification)

import { parseMpesaMessage, validateMpesaCode } from './js/index.js';

// User pastes confirmation message
const pastedMessage = document.getElementById('mpesaMsg').value;
const parsed = parseMpesaMessage(pastedMessage);

if (parsed.valid) {
    console.log('Receipt:', parsed.receipt);
    console.log('Amount:', parsed.amount);
    
    // Use receipt to confirm payment
    await confirmPayment(transactionId, parsed.receipt);
}

Track Order

import { trackOrder } from './js/index.js';

const result = await trackOrder('TP-KEN-XXXXX-ABC');

if (result.success) {
    const tx = result.transaction;
    console.log(`Status: ${tx.status}`);
    console.log(`Amount: KES ${tx.amount}`);
    console.log(`Created: ${new Date(tx.createdAt).toLocaleDateString()}`);
} else {
    console.log('Order not found');
}

Complete Form Example

Signup Form Integration

<form id="signupForm">
    <input type="text" id="name" required>
    <input type="email" id="email" required>
    <input type="tel" id="phone" required>
    <input type="password" id="password" required>
    <button type="submit">Sign Up</button>
</form>

<script type="module">
    import { signupUser } from './js/index.js';
    
    document.getElementById('signupForm').addEventListener('submit', async (e) => {
        e.preventDefault();
        
        const result = await signupUser(
            document.getElementById('email').value,
            document.getElementById('password').value,
            document.getElementById('name').value,
            document.getElementById('phone').value
        );
        
        if (result.success) {
            alert('Account created! Redirecting...');
            window.location.href = 'dashboard.html';
        } else {
            alert(result.error);
        }
    });
</script>
